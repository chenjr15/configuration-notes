```vimrc
" 显示行号
set number

" 语法高亮
syntax on

" 显示当前输入的命令
set showcmd

"显示当前的模式
set showmode

" 关闭鼠标支持, 便于复制
set mouse=


set encoding=utf-8

" 设置tab显示的宽度
set tabstop=2
" 缩进宽度
set shiftwidth=4

" 设置tab自动转换成空格
set expandtab
" 设置转换成空格的数量
set softtabstop=2

" 高亮当前行
set cursorline

"设置高亮搜索结果
set hlsearch 

"  显示状态栏 1 为多窗口显示 2为一直显示 0 为不显示
set laststatus=2


set  ruler
